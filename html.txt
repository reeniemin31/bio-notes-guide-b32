    <script>
        const chat = document.getElementById('chat-interface');
        const studyPage = document.getElementById('study-page');

        // Function to open the secret chat
        function openChat() {
            chat.style.display = 'flex';
            studyPage.style.filter = 'blur(5px)'; // Subtle blur for the background
        }

        // PANIC BUTTON: Press 'Escape' to hide instantly
        document.addEventListener('keydown', function(event) {
            if (event.key === "Escape") {
                chat.style.display = 'none';
                studyPage.style.filter = 'none';
            }
        });

        // Simple mock-up of sending a message
        const input = document.getElementById('user-input');
        const area = document.getElementById('message-area');

        input.addEventListener('keypress', function (e) {
            if (e.key === 'Enter' && input.value !== "") {
                const newMsg = document.createElement('div');
                newMsg.classList.add('bubble', 'sent');
                newMsg.innerText = input.value;
                area.appendChild(newMsg);
                input.value = "";
                area.scrollTop = area.scrollHeight;
            }
        });
    </script>
